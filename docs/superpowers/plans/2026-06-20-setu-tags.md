# Tags Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Let editors tag posts with free-form lowercase tags via chips + prefix-autocomplete, with suggestions sourced scale-safely from the content index.

**Architecture:** Tags are normalized lowercase strings stored in `metadata.tags` (draft) → frontmatter `tags: [...]` on publish. The existing content index is extended to project `tags` per entry, and a new `distinctTags(prefix, limit)` query (a pure `selectDistinctTags` helper shared by the memory + idb adapters, mirroring `runQuery`) feeds a `TagField` chip control in the editor MetaPanel.

**Tech Stack:** TypeScript, React 18, idb, Vitest + @testing-library/react.

## Global Constraints

- **Tags are normalized lowercase strings.** `normalizeTag` = lowercase, trim, drop punctuation, spaces/underscores→hyphen, collapse repeats, strip edges; returns `''` for empty/symbol-only. The UI rejects `''`.
- **Case-insensitive identity:** "React" and "react" are the same tag (`react`). No display-name storage.
- **Lifecycle:** applying a tag to a post is **draft metadata** (`metadata.tags`, a `string[]`), committed to frontmatter on publish. No immediate commit.
- **Suggestion source:** the content index (merges drafts + committed). `distinctTags(prefix, limit)` returns distinct tags whose value starts with the lowercased prefix, sorted ascending, capped at `limit`; empty prefix → first `limit` tags alphabetically.
- **Free-create + prefix autocomplete:** Enter or click a suggestion adds a chip; non-matching input + Enter free-creates.
- **`runQuery` is untouched** — tag *filtering* is deferred. `INDEX_VERSION` bumps 1→2 (row shape changed) so the persisted idb index auto-rebuilds.
- **Cloudflare-Pages-compatible + cost-safe:** pure functions + existing ports/adapters only; no new runtime deps.
- **Core tests colocate** as `src/**/*.test.ts` (the core vitest config discovers them — taxonomy precedent). **Admin tests live in `apps/admin/test/**`** (admin vitest only discovers `test/**`).
- Spec: `docs/superpowers/specs/2026-06-20-setu-tags-design.md`.

/** Normalize a raw tag to its canonical lowercase form: lowercase, trim, drop
 *  punctuation, spaces/underscores → hyphens, collapse repeats, strip edges.
 *  Returns '' for empty/symbol-only input (callers reject ''). */
export function normalizeTag(raw: string): string {
  return raw
    .toLowerCase()
    .trim()
    .replace(/[^\p{L}\p{N}\s_-]/gu, '')
    .replace(/[\s_]+/g, '-')
    .replace(/-+/g, '-')
    .replace(/^-+|-+$/g, '')
}

/** Normalize a list of raw tags: drop empties, dedupe preserving first-seen order. */
export function normalizeTags(raw: string[]): string[] {
  const out: string[] = []
  const seen = new Set<string>()
  for (const r of raw) {
    const t = normalizeTag(r)
    if (t && !seen.has(t)) {
      seen.add(t)
      out.push(t)
    }
  }
  return out
}
```

/** Tags from the live version: the draft's `tags` when a draft exists (even if
 *  empty — the editor may have cleared them), else the committed frontmatter's.
 *  Normalized + deduped; tolerant of absent/non-array. */
function tagsOf(draft: Draft | null, committedStr: string | null): string[] {
  if (draft) return normalizeFrom(draft.metadata['tags'])
  if (committedStr !== null) return normalizeFrom(parseMdoc(committedStr).frontmatter['tags'])
  return []
}

function normalizeFrom(raw: unknown): string[] {
  if (!Array.isArray(raw)) return []
  return normalizeTags(raw.filter((x): x is string => typeof x === 'string'))
}
```

import type { EntryIndexRow } from './types'

/** Distinct tags across rows whose value starts with the (lowercased) prefix,
 *  sorted ascending, capped at `limit`. Empty prefix → first `limit` tags. The
 *  single filter/sort impl, shared by every IndexPort adapter (cf. runQuery). */
export function selectDistinctTags(rows: EntryIndexRow[], prefix: string, limit: number): string[] {
  const p = prefix.toLowerCase().trim()
  const set = new Set<string>()
  for (const r of rows) for (const t of r.tags) if (t.startsWith(p)) set.add(t)
  return [...set].sort().slice(0, limit)
}
```

import { useEffect, useState } from 'react'
import { normalizeTag } from '@setu/core'
import { useIndex } from '../data/index-store'

const SUGGESTION_LIMIT = 8

export function TagField({
  selected,
  onChange,
  editable,
}: {
  selected: string[]
  onChange: (next: string[]) => void
  editable: boolean
}) {
  const index = useIndex()
  const [input, setInput] = useState('')
  const [suggestions, setSuggestions] = useState<string[]>([])

  useEffect(() => {
    const q = input.trim()
    if (q === '') {
      setSuggestions([])
      return
    }
    let cancelled = false
    const timer = setTimeout(() => {
      void index
        .distinctTags(q, SUGGESTION_LIMIT)
        .then((tags) => {
          if (!cancelled) setSuggestions(tags.filter((t) => !selected.includes(t)))
        })
        .catch(() => {
          if (!cancelled) setSuggestions([])
        })
    }, 150)
    return () => {
      cancelled = true
      clearTimeout(timer)
    }
  }, [input, index, selected])

  const add = (raw: string) => {
    const tag = normalizeTag(raw)
    setInput('')
    setSuggestions([])
    if (!tag || selected.includes(tag)) return
    onChange([...selected, tag])
  }

  const remove = (tag: string) => onChange(selected.filter((t) => t !== tag))

  return (
    <div className="tag-field">
      <div className="tag-chips">
        {selected.map((tag) => (
          <span key={tag} className="tag-chip">
            {tag}
            {editable && (
              <button type="button" className="tag-chip-x" aria-label={`Remove ${tag}`} onClick={() => remove(tag)}>
                ×
              </button>
            )}
          </span>
        ))}
      </div>
      <div className="tag-input-wrap">
        <input
          type="text"
          className="tag-input"
          placeholder="Add a tag"
          aria-label="Add a tag"
          value={input}
          disabled={!editable}
          onChange={(e) => setInput(e.target.value)}
          onKeyDown={(e) => {
            if (e.key === 'Enter') {
              e.preventDefault()
              add(input)
            }
          }}
        />
        {suggestions.length > 0 && (
          <ul className="tag-suggestions" role="listbox">
            {suggestions.map((tag) => (
              <li key={tag}>
                <button type="button" className="tag-suggestion" role="option" aria-selected={false} onClick={() => add(tag)}>
                  {tag}
                </button>
              </li>
            ))}
          </ul>
        )}
      </div>
    </div>
  )
}
```
